\section{Curvas}

\begin{dfn}
	$\forall i \in \inte{1,n}$, se define la función $ \rfunc{\pi_i}{\mathbb{R}^n}{\mathbb{R}}{(x_1, ..., x_n)}{x_i}$, llamada i-ésima proyección.
\end{dfn}

\begin{dfn}
	Sea $A \subseteq \mathbb{R}^ n $ y $\func{f}{A}{\mathbb{R}^n}$, a la función $\pi_j \circ f$, $\forall \, j \in \inte{1,m}$ le llamamos función componente jésima de $f$, denotada $f_j$. 
\end{dfn}


\begin{dfn}
	Sea $A \subseteq \mathbb{R}^n$ abierto, decimos que una función $\func{f}{A}{\mathbb{R}^n}$ es de clase $C^q \, \, (q \in \mathbb{N})$ si todas las derivadas parciales de $f$ hasta el orden $q$ existen y son continuas.
	
	Si $f$ es de clase $C^q \, \, \forall \, q \in \mathbb{N}$, decimos que $f$ es de clase $C^{\infty}$
\end{dfn}

\subsection{Curvas Parametrizadas}

\begin{dfn}
	Una \uwave{curva parametrizada} de clase $C^k \, \, (k \in \n \cup \{ \infty \})$ en $\rn \, \, (n\ge 2)$ es una función $\func{\sigma}{I}{\rn}$ de clase $C^k$, donde $I \subseteq \rr$ es un intervalo.

	A $\sigma (I)$ se le llama la traza de $\sigma$, y si $I=[a,b]$, con $\sigma (a)=\sigma(b)$, decimos que la curva $\sigma$ es cerrada. 
\end{dfn}


\begin{obs}

	Si $I$ no es un intervalo abieto y $k \ge 1$, significa que $\sigma$ se extiende a una función $\hat{\sigma}$ de clase $C^k $ definida en un intervalo abierto que contiene propiamente a $I$.

	Además, si $\sigma$ es cerrada de clase $C^k$, asumiremos que $\sigma' (a)= \sigma' (b), ..., \sigma ^{(k)} (a)= \sigma ^{(k)} (b)$ salvo que se especifique lo contrario.

\end{obs}

\subsection{Longitud de una Curva}

\begin{dfn}

	Si $\func{\sigma}{[a,b]}{\rn}$ es una curva parametrizada de clase $C^1$, se define
	 \[ \int_a ^b \sigma dt= \left( \int_a ^b \sigma_1 dt, ..., \int_a ^b \sigma_n dt   \right)  \]
	y se tiene que $\displaystyle \int_a ^b \sigma' dt= \left( \sigma_1(b)- \sigma_1(a), ..., \sigma_n(b)- \sigma_n(a)   \right) = \sigma(b)- \sigma(a) $

\end{dfn}


\begin{obs}

	Si $\func{\sigma}{[a,b]}{\rn}$ es una curva parametrizada de clase $C^1$, $\displaystyle \left\| \int_a ^b \sigma dt \right\| \le \int_a ^b \left \| \sigma \right \| dt $.

\end{obs}

\begin{dfn}

	Sea $P= \{ t_0, ..., t_k  \}$ una partición del intervalo $I=[a,b] \subseteq \rr$. Definimos la \uwave{norma de $P$} $\displaystyle \left\| P \right\| \coloneqq \text{máx} \{ |t_i-t_{i-1}| : i \in \inte{1,k}  \}$ 

\end{dfn}


\begin{dfn}

	Sea $\func{\sigma}{[a,b]}{\rn}$ una curva parametrizada y $P= \{ t_0, ..., t_k  \}$ una partición del intervalo $[a,b]$, se define la \uwave{longitud de la poligonal} que une los puntos $\sigma (t_i)$ como $\displaystyle L( \sigma ,P)= \sum _{i=1}^k \left\| \sigma (t_i)- \sigma (t_{i-1}) \right\|$


	Más aún, decimos que $\sigma$ es \uwave{rectificable} si $\displaystyle  \exists \, \, L( \sigma )= \lim_{\left \| P \right\| \to 0} L( \sigma ,P)  $ y $L( \sigma ) \in \rr$. 

	Si $\sigma$ es rectificable, a $L( \sigma )$ se le llama la \uwave{longitud de $\sigma$}.

\end{dfn}

\begin{thm}

	Sea $\func{\sigma}{[a,b]}{\rn}$ una curva parametrizada de clase $C^1$, entonces $\sigma$ es rectificable y \[ L( \sigma )= \int_a ^b \left\| \sigma' (t) \right\| dt   \]

\end{thm}



\begin{proof}

	Sea $\epsilon > 0$, como $ \sigma $ es de clase $C^1$, $ \sigma'$ es continua sobre $[a,b]$ compacto; luego, $\exists \, \delta > 0 \, : \, \forall \, t, s \in [a,b]$ con $|t-s| < \delta$, $\displaystyle \left\| \sigma' (t)- \sigma' (s) \right\| < \frac{\epsilon}{b-a}  $.

	Sea $P=\{ t_0, ..., t_k \}$ una partición de $[a,b] \, : \, \left\| P \right\| < \delta $. Primeramente, note que $ \forall \, i \in \inte{1,k}$, $\displaystyle \left\| \sigma (t_i)- \sigma (t_{i-1})  \right\|= \left\| \int_{t_{i-1}}^{t_i} \sigma' (t) dt  \right\| \le \int_{t_{i-1}}^{t_i} \left\|  \sigma' (t) \right\|  dt  $.

	Lo cual implica que $\displaystyle L( \sigma ,P)= \sum_{i=1}^k \left\| \sigma (t_i)- \sigma (t_{i-1}) \right\| \le \sum_{i=1}^k \int_{t_{i-1}} ^{t_i} \left\|  \sigma' (t) \right\|  dt = \int_a ^b  \left\|  \sigma' (t) \right\|  dt  $. (*)
	
	
	
	Por otra parte, 
	\begin{align*}
		\sigma(t_i)-\sigma(t_{i-1}) &= \int_{t_{i-1}}^{t_i} \sigma' (t)dt =\int_{t_{i-1}}^{t_i} (\sigma' (s)+\sigma' (t)-\sigma' (s))dt \\
		&=\int_{t_{i-1}}^{t_i} \sigma' (s)dt + \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt = \sigma'(s) (t_i-t_{i-1})+ \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt
	\end{align*}

	De esto se sigue que \[ \left\| \sigma(t_i)-\sigma(t_{i-1}) - \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt \right\| = \left\| \sigma'(s) (t_i-t_{i-1}) \right\| \]
	
	Pero \[ \left\| \sigma(t_i)-\sigma(t_{i-1}) \right|\ + \left\| \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt \right\| \ge \left\| \sigma(t_i)-\sigma(t_{i-1}) - \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt \right\| \]
	
	Con lo cual,
	\begin{align*}
		\left\| \sigma(t_i)-\sigma(t_{i-1}) \right\| &\ge \left\| \sigma'(s) (t_i-t_{i-1}) \right\| - \left\| \int_{t_{i-1}}^{t_i} (\sigma' (t)-\sigma' (s))dt \right\| \\
		&\ge \left\| \sigma'(s) (t_i-t_{i-1}) \right\| -  \int_{t_{i-1}}^{t_i} \left\| (\sigma' (t)-\sigma' (s)) \right\| dt | \\
		&\ge \left\| \sigma'(s) \right\| (t_i-t_{i-1}) - \frac{\epsilon}{b-a} (t_i-t_{i-1})
	\end{align*}
	
	Esto es
	\[
		\left\| \sigma(t_i)-\sigma(t_{i-1}) \right\| = \int_{t_{i-1}}^{t_i} \frac{\left\| \sigma(t_i)-\sigma(t_{i-1}) \right\|}{t_i-t_{i-1}} \ge \int_{t_{i-1}}^{t_i} \left\| \sigma'(s) \right\| ds - \frac{\epsilon}{b-a} (t_i-t_{i-1})
	\]
	
	Por lo tanto
	\[ \sum_{i=1}^k  \left\| \sigma(t_i)-\sigma(t_{i-1}) \right\| \ge \sum_{i=1}^k \int_{t_{i-1}}^{t_i} \left\| \sigma'(s) \right\| ds - \frac{\epsilon}{b-a}  \sum_{i=1}^k (t_i-t_{i-1}) \]
	
	Finalmente, de (*)
	
	\[ \left| \int_{a}^{b} \left\| \sigma '(s) \right\| ds - L(\sigma, P) \right| \le \epsilon \]

\end{proof}

\subsection{Reparametrización}

\begin{dfn}
	Sean $I, J \subseteq \rr $ intervalos abiertos, un difeormismo de $I$ en $J$ es una función biyectiva $\func{f}{I}{J}$ tal que $f$ y $f^{-1}$ son diferenciables sobre $I$ y $J$, respectivamente.
\end{dfn}

\begin{obs}
	Si $\func{f}{I}{J}$, $\func{g}{J}{L}$ son difeomorfismos, entonces $\func{g \circ f}{I}{L}$ es un difeomorfismo.
\end{obs}
